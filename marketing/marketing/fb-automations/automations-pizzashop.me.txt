Making the ‚Äúrevenue _from _pdf( )‚Äù helper bullet-proof
Your original helper grabs the first number it sees, which works only if ‚ÄúNet Sales‚Äù is the very first dollar figure on every statement. Most PDFs put many figures before the sales line (dates, page numbers, cash balances, etc.), so we need to anchor the capture to keywords such as ‚ÄúNet Sales‚Äù, ‚ÄúSales‚Äù, ‚ÄúTotal Sales‚Äù, ‚ÄúSales Revenue‚Äù, ‚ÄúNet Revenue‚Äù, while avoiding near-matches like ‚ÄúCost of Sales‚Äù.

Below is a drop-in replacement that:

Scans each page line-by-line (digital text first, OCR fallback).

Uses an anchored, case-insensitive regex to look only at lines that contain target labels.

Falls back to simple table parsing (via pdfplumber.extract_table) if no line match is found.

import re, pdfplumber, pytesseract
from pdf2image import convert_from_path
from pathlib import Path
from typing import Optional

# ‚îÄ‚îÄ 1. Regex for a labelled sales line ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ
LABELLED_SALES_RE = re.compile(
    r"(?i)\b(?:net\s*)?(?:sales|total\s*sales|sales\s*revenue|net\s*revenue)\b"  # keywords
    r"[^\d\-]{0,40}"                      # up to 40 non-digit chars (tabs, dots, etc.)
    r"(-?\$?\s*[0-9][0-9,]*\.\d{2})"      # capture the dollar amount (incl. negatives)
)

# optional second defence: a plain dollar-amount for later fallback
ANY_AMOUNT_RE = re.compile(r"\$?\s*([0-9][0-9,]*\.\d{2})")


def _find_sales_in_text(text: str) -> Optional[float]:
    """Return sales figure from a block of text or None."""
    for line in text.splitlines():
        m = LABELLED_SALES_RE.search(line)
        if m:
            return float(m.group(1).replace("$", "").replace(",", "").strip())
    return None


def revenue_from_pdf(fp: Path) -> Optional[float]:
    """Extract Net/Total Sales from a PDF‚Äîdigital first, OCR fallback."""
    # ‚ë† Digital text search
    with pdfplumber.open(fp) as pdf:
        for page in pdf.pages:
            txt = page.extract_text() or ""
            amount = _find_sales_in_text(txt)
            if amount is not None:
                return amount

            # ‚ë° If labelled line not found, try table rows
            tables = page.extract_tables()
            for tbl in tables:
                for row in tbl:
                    row_join = " ".join(str(c) for c in row if c)
                    amount = _find_sales_in_text(row_join)
                    if amount is not None:
                        return amount

    # ‚ë¢ OCR fallback for scanned PDFs
    images = convert_from_path(fp, dpi=300)
    for img in images:
        ocr_txt = pytesseract.image_to_string(img)
        amount = _find_sales_in_text(ocr_txt)
        if amount is not None:
            return amount

    # ‚ë£ As a last-ditch effort, return the *largest* amount on page 1
    #    (often‚Äîbut not always‚ÄîSales for single-page P&L statements)
    with pdfplumber.open(fp) as pdf:
        nums = ANY_AMOUNT_RE.findall(pdf.pages[0].extract_text() or "")
        if nums:
            return max(float(n.replace(",", "")) for n in nums)

    return None      # let caller decide what to do next

# fb_post.py  ‚Äì publish a single image to FB + IG
import os, requests, time
from dotenv import load_dotenv
load_dotenv()

PAGE_ID   = os.getenv("FB_PAGE_ID")
IG_ID     = os.getenv("IG_BUSINESS_ID")      # linked IG
TOKEN     = os.getenv("FB_PAGE_TOKEN")       # long-lived

def publish_to_fb(img_path, caption):
    # 1. upload media
    r = requests.post(
        f"https://graph.facebook.com/v19.0/{PAGE_ID}/photos",
        data={"caption": caption, "published": "false", "access_token": TOKEN},
        files={"source": open(img_path, "rb")}
    )
    media_id = r.json()["id"]
    # 2. publish
    requests.post(
        f"https://graph.facebook.com/v19.0/{PAGE_ID}/media_publish",
        data={"creation_id": media_id, "access_token": TOKEN}
    )

def publish_to_ig(img_path, caption):
    # similar two-step flow for IG
    media = requests.post(
        f"https://graph.facebook.com/v19.0/{IG_ID}/media",
        data={"caption": caption, "image_url": img_path, "access_token": TOKEN}
    ).json()
    time.sleep(5)  # wait for processing
    requests.post(
        f"https://graph.facebook.com/v19.0/{IG_ID}/media_publish",
        data={"creation_id": media['id'], "access_token": TOKEN}
    )
