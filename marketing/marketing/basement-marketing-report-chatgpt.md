def run_scenario(spend: float) -> dict[str, float]:

    """Return key metrics for a single monthly spend level."""



    channel_stats: Dict[str, Dict[str, float]] = {}

    for ch, share in ALLOCATION.items():

        click_budget = spend * share

        clicks = click_budget / CPC[ch]

        conversions = clicks * CVR[ch]

        channel_stats[ch] = {"clicks": clicks, "conversions": conversions}



    total_clicks = sum(v["clicks"] for v in channel_stats.values())

    total_conversions = sum(v["conversions"] for v in channel_stats.values())



    revenue = total_conversions * AOV

    gross_profit = revenue * GROSS_MARGIN

    profit_after_ads = gross_profit - spend



    return {

        "clicks": total_clicks,

        "conversions": total_conversions,

        "revenue": revenue,

        "gross_profit": gross_profit,

        "profit_after_ads": profit_after_ads,

        "channel_breakdown": channel_stats,

    }

def get_active_cvr() -> Dict[str, float]:

    """Return the CVR table according to PROJECTION_MODE."""

    return CVR_BEST if PROJECTION_MODE == "best" else CVR_BASE





def run_paid_channel_block(spend: float, cvr_tbl: Dict[str, float]) -> Dict[str, Dict[str, float]]:

    """Return clicks & conversions for Google & Meta given paid‚Äëad spend."""

    stats: Dict[str, Dict[str, float]] = {}

    for ch, share in ALLOCATION.items():

        click_budget = spend * share

        clicks = click_budget / CPC[ch]

        conversions = clicks * cvr_tbl[ch]

        stats[ch] = {"clicks": clicks, "conversions": conversions}

    return stats





def run_seo_block(month_idx: int, seo_spend: float) -> Dict[str, float]:

    """Return clicks, conversions and ROI for the SEO slice in a month."""

    if month_idx < SEO_START_MONTH:

        return {"clicks": 0.0, "conversions": 0.0, "revenue": 0.0, "roi": 0.0}



    # Assume the first 3 months are ramp‚Äëup (no traffic benefit)

    ramp_up_months = 3

    if month_idx < SEO_START_MONTH + ramp_up_months:

        clicks = 0.0

    else:

        clicks = seo_spend * SEO_CLICKS_PER_USD



    conversions = clicks * SEO_CVR

    revenue = conversions * AOV

    roi = (revenue - seo_spend) / seo_spend if seo_spend else 0

    return {"clicks": clicks, "conversions": conversions, "revenue": revenue, "roi": roi}
